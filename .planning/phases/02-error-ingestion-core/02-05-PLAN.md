<!-- OTLP JSON format from RESEARCH.md -->
OTLP LogRecord structure:
```json
{
  "resourceLogs": [{
    "resource": { "attributes": [{"key":"service.name","value":{"stringValue":"..."}}] },
    "scopeLogs": [{
      "logRecords": [{
        "timeUnixNano": "1709467200000000000",
        "severityNumber": 17,
        "severityText": "ERROR",
        "body": {"stringValue": "..."},
        "attributes": [
          {"key":"exception.type","value":{"stringValue":"TypeError"}},
          {"key":"exception.message","value":{"stringValue":"..."}},
          {"key":"exception.stacktrace","value":{"stringValue":"..."}}
        ]
      }]
    }]
  }]
}
```

OTLP SeverityNumber mapping: 17-20=error, 21-24=fatal, 13-16=warning, 9-12=info

Generic JSON API schema:
```json
{
  "message": "...", "level": "error", "environment": "production",
  "exception": {
    "type": "TypeError", "value": "...",
    "stacktrace": [{"filename":"...","function":"...","lineno":42,"in_app":true}]
  },
  "tags": {}, "extra": {}, "release": "...", "server_name": "..."
}
```

JSON-only is the correct approach for Phase 2.

**Rate limit response helper (shared across OTLP handlers):**

fn otlp_rate_limit_response(retry_after :: String) -> Response do
  # Use HTTP.response_with_headers for proper rate limit headers.
  # Confirmed pattern from server/src/auth/session.mpl:59.
  let headers = %{

**OTLP Logs handler:**
```
pub fn handle_otlp_logs(pool :: PoolHandle, request) -> Response do
  # 1. Content-Type check (415 for protobuf)
  # 2. Auth: extract_and_validate_api_key (Bearer token)
  # 3. Rate limit check -- if not allowed: return otlp_rate_limit_response(retry_after)
  # 4. Parse body as JSON
  # 5. Navigate: resourceLogs[*].scopeLogs[*].logRecords[*]
  # 6. For each LogRecord:
  #    a. Check severityNumber >= 17 (ERROR level) -- only process errors
  #       For non-error log records: skip (Phase 2 is error ingestion only)
  #    b. Map severityNumber to level: 17-20="error", 21-24="fatal", 13-16="warning"
  #    c. Extract from attributes array (key-value pairs):
  #       - exception.type -> exception_type
  #       - exception.message -> exception_value
  #       - exception.stacktrace -> raw stacktrace string
  #    d. Extract resource attributes:
  #       - service.name -> server_name
  #       - deployment.environment -> environment (default "production")
  #    e. Parse OTLP stacktrace string into frames JSON:
  #       OTLP stacktrace is a plain text string like:
  #       "TypeError: Cannot read...\n    at getUsers (app/routes/users.js:42:12)\n    at Layer.handle (...)"
  #       Parse each "at <function> (<file>:<line>:<col>)" into structured frame JSON
  #    f. body.stringValue -> message
  #    g. timeUnixNano -> convert to ISO timestamp string
  #       (divide by 1_000_000_000 to get seconds, format as timestamp)
  #    h. Generate event_id with Crypto.uuid4() (OTLP doesn't provide one)
  #    i. Scrub, fingerprint, upsert issue, insert event (same pipeline as Sentry handler)
  # 7. Return: HTTP 200 {"partialSuccess":{}} (OTLP spec response)
end
```

pub fn handle_otlp_traces(pool :: PoolHandle, request) -> Response do

```
pub fn handle_generic_event(pool :: PoolHandle, request) -> Response do
  # 1. Auth: extract_and_validate_api_key (Bearer token)
  # 2. Rate limit check -- if not allowed: return generic_rate_limit_response(retry_after)
  # 3. Parse JSON body:
  #    Required: message (String)
  #    Optional with defaults:
  #      level: "error"
  #      environment: "production"
  #      exception.type: "" (extract from exception object if present)
  #      exception.value: "" (extract from exception object if present)
  #      exception.stacktrace: [] (extract frames array if present)
  #      tags: {}
  #      extra: {}
  #      release: ""
  #      server_name: ""
  #
  # 4. Normalize stacktrace frames to JSON string:
  #    Input frames have: filename, function, lineno (optional), in_app (optional)
  #    Normalize same as other handlers: strip lineno, mark in_app
  #
  # 5. Generate event_id with Crypto.uuid4()
  # 6. Scrub all fields
  # 7. Compute fingerprint
  # 8. Upsert issue, insert event
  # 9. Return: HTTP 200 {"id": "<event_id>"}
  #
  # Validation:
  # - If body is not valid JSON: return 400 {"error": "invalid JSON"}
  # - If message is missing: return 400 {"error": "message is required"}
end
```
